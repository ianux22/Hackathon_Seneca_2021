import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

# a4_dims = (11.7, 8.27)
# fig, ax = pyplot.subplots(figsize=a4_dims)

for p in j.patches:
    width = p.get_width()
    plt.text(80+p.get_width(), p.get_y()+0.55*p.get_height(),
             '{:1.2f}'.format(width),
             ha='center', va='center')
    
plt.xlabel("Count of Rare Diseases", size=14)
